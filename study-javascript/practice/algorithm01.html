// 나의 답
function solution(a){
    let sum = 0;
    a.forEach(e => {
        sum += e;
    });
    let avg = sum / a.length;
    return avg;
}
// 다른 풀이
function average(array){
  return array.reduce((a, b) => a + b) / array.length;
}

// 나의 답
function solution(arr){
  var answer = [];
  arr.forEach((e) => {
    if(answer[answer.length - 1] != e){
        answer.push(e);
    }
  });
  return answer;
}
// 다른 풀이
function solution(arr){
    return arr.filter((val,index) => val != arr[index+1]);
}
