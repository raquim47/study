1. 1부터 10까지 출력

01. while문


  {
    let num = 1;
  
    while(num <= 10){
        document.write(num);
        num++;
      }
  }

02. for문


  {
    for(let i=1; i<=10; i++){
        document.write(i);
    }
  }

03. for문 홀수, 짝수


  {
    //짝수
    for(let i=1; i<=10; i++){
        if( i % 2 == 0 ){
            console.log(i);
        }
    }
  
    //홀수
    for(let i=1; i<=10; i++){
        if( i % 2 != 0 ){
            console.log(i);
        }
    }
  
    //다른 방법1 if문 중괄호 생략
    for(let i=1; i<=10; i++){
        if( i % 2 == 0 ) console.log(i);
    }
    
    //다른 방법2
    for(let i=2; i<=10; i+=2){
        console.log(i);
    }
  }

2. 구구단 출력

01. 2단부터 5단까지 출력 for문


{
  for(let i=2; i<=5; i++){
      for(let j=1; j<=9; j++){
          document.write(i + " * " + j + " = " +  i*j + "
");
      }
      document.write("=============
");
  }
}

02. 2단부터 5단까지 출력 while문


{
  let i = 2, j = 1;

  while(i <= 5){
      while(j <= 9){
          document.write(i + " * " + j + " = " + i*j + "
");
          j++;
      }
      j = 1;
      i++;
      document.write("=============
");
  }
}

03. 사용자가 원하는 구구단을 입력해서 출력


{
  let num, result;

  //버튼을 클릭하면 경고창을 띄어주세요.
  document.querySelector(".pBtn").addEventListener("click", () => {
      num = prompt("출력할 구구단 숫자를 입력해주세요!!", "2");
      document.querySelector(".result > div").classList.add("active"); // 버튼을 클릭하면 결과창이 무조건 열리게
      
      for(let i = 1; i <= 9; i++){
          result = num + " * " + i + " = " + (num*i) + "
";
          // console.log(result);
          document.querySelector(".result > div").innerHTML += result;
      }
  });
}
