06. switch문)

{
  const num = 100;
  
  switch(num){
    case 100 :
        document.write("실행되었습니다(num==100)");
    break;
    case 200 :
        document.write("실행되었습니다(num==200)");
    break;
    case 300 :
        document.write("실행되었습니다(num==300)");
    break;
    case 400 :
        document.write("실행되었습니다(num==400)");
    break;
    default :

  }

11. break문

{
  //while문을 이용해서 0~20까지 출력
  let num = 1;
  while(num <= 20){
    document.write(num);
    if(num == 10){
        break;
    }
    num++;
  }
  
  //for문을 이용해서 0~20 출력
  for(let i = 1; i <= 20; i++){
    document.write(i);
    if(i == 10){
        break;
    }
  }
  }

12. continue문

{
  let num = 0;
  while(num < 10){
    num++;
    if ( num == 5){
        continue;
    }
    document.write(num);
  }
  
  for(let i = 1; i<=10; i++){
    if( i == 5){
        continue;
    }
    document.write(i);
