07. 배열 : 데이터 불러오기 : forEach()

{
    const num = [100, 200, 300, 400, 'hong'];

    document.write(num[0]);
    document.write(num[1]);
    document.write(num[2]);
    document.write(num[3]);
    document.write(num[4]);

    //for문
    for( i=0; i<num.length; i++ ){
        document.write(num[i]);
    }

    //forEach() 매서드 
    //배열명.forEach(콜백함수(파라미터1, 파라미터2, 파라미터3){
    //    자바스크립트 코드;
    //});
    //파라미터1 : 배열의 값
    //파라미터2 : 배열의 인덱스 번호
    //파라미터3 : 배열 자체

    //forEach()
    num.forEach(function(el){
        document.write(el);
    });

    //forEach(element, index, array)
    num.forEach(function(element, index, array){
        document.write(element);
        document.write(index);
        document.write(array);
    });

    // 값: (el, index, array 자체) el 개수만큼 반복해서 출력
}

09. 배열 : 데이터 불러오기 : for in

{
    const num = [100, 200, 300, 400, 'hong'];

    const book = {color: 'red', price: 1200 };

    for( let i in num ){
        document.write(i);
    }

    for( let i in num ){
        document.write(num[i]);
    }

    //for in 반복문은 객체에 주로 사용
    //key 값에 객체의 key가 들어가게 됨.
    for(let key in book ) {
      document.write(num[key]);
    }

    for(let key in book ) {
      document.write(key);
    }
}

11. 배열 : 데이터 불러오기 : 펼침 연산자(spread Operator)

{
    // spread Operator : array에 붙이면 대괄호를 제거. 즉 배열을 펼쳐줌. 
    // * 문자에 붙일 경우 글자마다 쪼개서 펼침
    // ex) 'hello' -> 'h' 'e 'l' 'l'' 'o'       
    const num = [100, 200, 300, 400, 500]
    document.write(num);
    document.write(num[0], num[1], num[2], num[3], num[4]);
    document.write(...num);
}
